import java.util.Scanner;

public class Menu {
    void go() {
        boolean flag = true;
        while (flag) {
            System.out.println("Choose the option: ");
            System.out.println("1)Lokomotiv");
            System.out.println("2)Kupe");
            System.out.println("3)Plaskart");
            System.out.println("4)Cargo");
            System.out.println("5)Exit");
            Scanner sc = new Scanner(System.in);
            int choice = sc.nextInt();
            switch (choice) {
                case 1:
                    Lok lok = new Lok("Engine", 999, 50, 1111, 0);
                    break;
                case 2:
                    Kupe kupe = new Kupe("Passenger", 500, 30, 100, 20);
                    break;
                case 3:
                    Pla4 pla4kart = new Pla4("Passenger", 700, 50, 100, 20);
                    break;
                case 4:
                    Load load = new Load("Load", 10000, 333, 0, 0);
                    break;
                case 5:
                    System.out.println("Exiting");
                    flag = false;
                    break;
            }
        }
    }
}
